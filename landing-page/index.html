Developer Experience

Simple Integration, Powerful Coordination

Get started in minutes with our Python SDK. Structured AI coordination is just a few lines of code away. 

# Install the AI Triad Model package
pip install ai-triad-model

# Initialize the multi-agent framework
from triad import TriadFramework, TriadDeps

# Create structured framework
framework = TriadFramework()
deps = TriadDeps.from_environment()

# Your AI now operates with structured oversight
async def structured_ai_decision(user_request: str):
    """Process requests through systematic oversight"""
    
    # Request goes through all four specialized agents
    result = await framework.process_request(
        request=user_request,
        deps=deps,
        require_consensus=True
    )
    
    return result
# Configure the four specialized agents
from triad.agents import (
    PlannerAgent,    # Strategic Planning
    ExecutorAgent,   # Task Execution  
    EvaluatorAgent,  # Quality Review
    OverwatchAgent   # System Monitoring
)

# Initialize agents with specialized roles
planner = PlannerAgent(
    agent_role="strategic_planning",
    system_accountability=True
)

executor = ExecutorAgent(
    agent_role="task_execution", 
    structured_oversight=True
)

evaluator = EvaluatorAgent(
    agent_role="quality_review",
    compliance_checking=True
)

overwatch = OverwatchAgent(
    agent_role="system_monitoring",
    crisis_management=True
)
# Structured coordination process
async def agent_coordination_flow(request: str):
    """Full four-agent coordination process"""
    
    # 1. Planning: Create strategy/plan
    plan = await planner.create_strategy(request)
    
    # 2. Execution: Implement the plan  
    execution = await executor.implement_plan(plan)
    
    # 3. Evaluation: Review for quality
    review = await evaluator.quality_review(execution)
    
    # 4. Oversight: Final approval
    oversight = await overwatch.final_approval(review)
    
    # Return systematically validated result
    return {
        "result": oversight.approved_result,
        "system_compliance": True,
        "transparency_log": oversight.audit_trail,
        "coordination_method": "Four-Agent Framework"
    }
