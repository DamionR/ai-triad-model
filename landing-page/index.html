Developer Experience

Simple Integration, Powerful Governance

Get started in minutes with our Python SDK. Democratic AI governance is just a few lines of code away. 

# Install the AI Triad Constitutional package
pip install ai-triad-constitutional

# Initialize the Westminster Parliamentary system
from triad import ConstitutionalFramework, TriadDeps

# Create constitutional framework
framework = ConstitutionalFramework()
deps = TriadDeps.from_environment()

# Your AI now operates under democratic governance
async def democratic_ai_decision(user_request: str):
    """Process requests through constitutional oversight"""
    
    # Request goes through all four constitutional agents
    result = await framework.process_democratic_request(
        request=user_request,
        deps=deps,
        require_consensus=True
    )
    
    return result
# Configure the four constitutional agents
from triad.agents import (
    PlannerAgent,    # Legislative Authority
    ExecutorAgent,   # Executive Authority  
    EvaluatorAgent,  # Judicial Authority
    OverwatchAgent   # Crown Authority
)

# Initialize agents with constitutional roles
planner = PlannerAgent(
    constitutional_authority="legislative",
    democratic_accountability=True
)

executor = ExecutorAgent(
    constitutional_authority="executive", 
    parliamentary_oversight=True
)

evaluator = EvaluatorAgent(
    constitutional_authority="judicial",
    constitutional_review=True
)

overwatch = OverwatchAgent(
    constitutional_authority="crown",
    crisis_management=True
)
# Democratic decision-making process
async def parliamentary_procedure(request: str):
    """Full Westminster parliamentary process"""
    
    # 1. Legislative: Create policy/plan
    plan = await planner.draft_policy(request)
    
    # 2. Executive: Implement the plan  
    execution = await executor.implement_policy(plan)
    
    # 3. Judicial: Review for compliance
    review = await evaluator.constitutional_review(execution)
    
    # 4. Crown: Final oversight
    oversight = await overwatch.final_approval(review)
    
    # Return democratically validated result
    return {
        "result": oversight.approved_result,
        "constitutional_compliance": True,
        "transparency_log": oversight.audit_trail,
        "democratic_legitimacy": "Westminster Parliamentary"
    }
